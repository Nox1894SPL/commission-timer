<script>

  var totalSeconds = 180;

  var remaining = 180;

  var interval = null;

  var running = false;

  var audioCtx = null;


  function getCtx() {

    if (!audioCtx) audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    if (audioCtx.state === 'suspended') audioCtx.resume();

    return audioCtx;

  }


  function playBeep() {

    var ctx = getCtx();

    var now = ctx.currentTime;

    [0, 0.4, 0.8].forEach(function(offset) {

      var osc = ctx.createOscillator();

      var gain = ctx.createGain();

      osc.connect(gain);

      gain.connect(ctx.destination);

      osc.type = 'sine';

      osc.frequency.value = 880;

      gain.gain.setValueAtTime(0, now + offset);

      gain.gain.linearRampToValueAtTime(0.7, now + offset + 0.02);

      gain.gain.exponentialRampToValueAtTime(0.001, now + offset + 0.3);

      osc.start(now + offset);

      osc.stop(now + offset + 0.4);

    });

  }


  function fmt(s) {

    return Math.floor(s / 60) + ':' + (s % 60 < 10 ? '0' : '') + (s % 60);

  }


  var display  = document.getElementById('timer-display');

  var statusEl = document.getElementById('status');

  var mainBtn  = document.getElementById('main-btn');

  var bar      = document.getElementById('progress-bar');


  function updateDisplay() {

    display.textContent = fmt(remaining);

    bar.style.width = ((remaining / totalSeconds) * 100) + '%';

    if (remaining === 0) {

      display.style.color = '#E24B4A';

      bar.style.background = '#E24B4A';

      display.classList.add('flash');

      statusEl.textContent = 'Time is up!';

      statusEl.style.color = '#E24B4A';

      mainBtn.textContent = 'Start';

      mainBtn.className = 'btn btn-start';

      running = false;

      clearInterval(interval);

      interval = null;

      playBeep();

    } else if (remaining <= 30) {

      display.style.color = '#BA7517';

      bar.style.background = '#EF9F27';

      statusEl.textContent = 'Less than 30 seconds remaining';

      statusEl.style.color = '#BA7517';

    } else {

      display.style.color = '#222';

      bar.style.background = '#1D9E75';

      statusEl.textContent = 'Timer running...';

      statusEl.style.color = '#888';

    }

  }


  function handleMain() {

    getCtx();

    if (remaining === 0) return;

    if (running) {

      clearInterval(interval); interval = null; running = false;

      mainBtn.textContent = 'Resume';

      statusEl.textContent = 'Paused';

      statusEl.style.color = '#888';

    } else {

      running = true;

      mainBtn.textContent = 'Pause';

      statusEl.textContent = 'Timer running...';

      interval = setInterval(function() { remaining--; updateDisplay(); }, 1000);

    }

  }


  function handleReset() {

    clearInterval(interval); interval = null; running = false;

    remaining = totalSeconds;

    display.style.color = '#222';

    display.classList.remove('flash');

    bar.style.background = '#1D9E75';

    bar.style.width = '100%';

    display.textContent = fmt(remaining);

    mainBtn.textContent = 'Start';

    mainBtn.className = 'btn btn-start';

    statusEl.textContent = 'Ready — press Start to begin';

    statusEl.style.color = '#888';

  }


  function setDuration(mins, pid) {

    document.querySelectorAll('.pbtn').forEach(function(b){ b.classList.remove('active'); });

    document.getElementById(pid).classList.add('active');

    totalSeconds = mins * 60;

    remaining = totalSeconds;

    clearInterval(interval); interval = null; running = false;

    display.style.color = '#222';

    display.classList.remove('flash');

    bar.style.background = '#1D9E75';

    bar.style.width = '100%';

    display.textContent = fmt(remaining);

    mainBtn.textContent = 'Start';

    mainBtn.className = 'btn btn-start';

    statusEl.textContent = 'Ready — press Start to begin';

    statusEl.style.color = '#888';

  }
